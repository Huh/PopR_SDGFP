model{
  #  Priors
  comp[1:3] ~ ddirch(c(1, 1, 1))
			
  #  Likelihood
  for(i in 1:nobs){
    y[i,] ~ dmulti(comp[], n[i])
  }
	
  #  Derive sex and age ratios
  mf <- comp[3]/comp[2]
  yf <- comp[1]/comp[2]

  #  Rename comp for easier handling
  paf <- comp[2]
  pam <- comp[3]
  py <- comp[1]
}
